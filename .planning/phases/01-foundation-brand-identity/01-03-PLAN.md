Create effects tokens in W3C DTCG format.

**Border Radius Scale:**
```
borderRadius:
  none: 0px
  sm: 2px      (subtle rounding)
  base: 4px    (buttons, inputs)
  md: 6px      (cards)
  lg: 8px      (larger cards, modals)
  xl: 12px     (pills, large elements)
  2xl: 16px    (feature cards)
  3xl: 24px    (large decorative)
  full: 9999px (circular/pill shapes)
```

**Shadow System (Elevation):**
Modern, subtle shadows for B2B professional look:
```
shadow:
  none: "none"
  sm: "0 1px 2px 0 rgb(0 0 0 / 0.05)"
  base: "0 1px 3px 0 rgb(0 0 0 / 0.1), 0 1px 2px -1px rgb(0 0 0 / 0.1)"
  md: "0 4px 6px -1px rgb(0 0 0 / 0.1), 0 2px 4px -2px rgb(0 0 0 / 0.1)"
  lg: "0 10px 15px -3px rgb(0 0 0 / 0.1), 0 4px 6px -4px rgb(0 0 0 / 0.1)"
  xl: "0 20px 25px -5px rgb(0 0 0 / 0.1), 0 8px 10px -6px rgb(0 0 0 / 0.1)"
  2xl: "0 25px 50px -12px rgb(0 0 0 / 0.25)"
  inner: "inset 0 2px 4px 0 rgb(0 0 0 / 0.05)"

elevation:
  card: (references shadow.base)
  cardHover: (references shadow.md)
  dropdown: (references shadow.lg)
  modal: (references shadow.xl)
  tooltip: (references shadow.md)
```
